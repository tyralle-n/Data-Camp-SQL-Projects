SELECT public.student_performance.attendance, public.student_performance.hours_studied, public.student_performance.sleep_hours, public.student_performance.tutoring_sessions, 
	DENSE_RANK() OVER(ORDER BY public.student_performance.exam_score DESC) AS exam_rank
FROM public.student_performance
	ORDER BY exam_rank
	LIMIT 30;
